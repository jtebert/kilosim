19 class CollisionBoxes
20 {
21 private:
22  const int cddx[9] = {0, -1, -1, 0, 1, 1, 1, 0, -1};
23  const int cddy[9] = {0, 0, -1, -1, -1, 0, 1, 1, 1};
24  const int PSIZE = 4;
25  typedef std::vector<int> ivec;
26  ivec agent_positions;
27  ivec cells_used;
28  double diameter; //Collision diameter
29  int bwidth; //Width in bins
30  int bheight; //Height in bins
31 
32 public:
33  CollisionBoxes() = default;
34 
35  CollisionBoxes(const double width0, const double height0, const double diameter0)
36  {
37  diameter = diameter0;
38  bwidth = std::ceil(width0 / diameter);
39  bheight = std::ceil(height0 / diameter);
40 
41  agent_positions.resize(PSIZE * bwidth * bheight, -1);
42  }
43 
44  template <class T>
45  void update(const std::vector<T> &agents)
46  {
47  //Clear bins of their occupants
48  for (const auto &p : cells_used)
49  agent_positions[p] = -1;
50  cells_used.clear();
51 
52  for (unsigned int a = 0; a < agents.size(); a++)
53  {
54  const int binx = agents[a].x / diameter;
55  const int biny = agents[a].y / diameter;
56  const int idx0 = PSIZE * (biny * bwidth + binx);
57  int idx = idx0;
58  for (; idx <= idx0 + PSIZE; idx++)
59  if (agent_positions[idx] == -1)
60  break;
61  assert(idx != idx0 + PSIZE);
62  agent_positions[idx] = a;
63  cells_used.emplace_back(idx);
64  }
65  }
66 
67  template <class F>
68  void considerNeighbours(const double x, const double y, F func) const
69  {
70  const int cbinx = x / diameter;
71  const int cbiny = y / diameter;
72 
73  for (unsigned int nbi = 0; nbi <= 8; nbi++)
74  {
75  const int binx = cbinx + cddx[nbi];
76  const int biny = cbiny + cddy[nbi];
77 
78  if (binx < 0 || biny < 0 || binx == bwidth || biny == bheight)
79  continue;
80 
81  const auto idx0 = &agent_positions[PSIZE * (biny * bwidth + binx)];
82  for (auto idx = idx0; idx < idx0 + PSIZE; idx++)
83  {
84  if (*idx == -1)
85  continue;
86  //If func returns false, that means it doesn't want to look at any more
87  //neighbours
88  if (!func(*idx))
89  return;
90  }
91  }
92  }
93 };
